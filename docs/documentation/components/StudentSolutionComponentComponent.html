import { Component, OnInit } from '@angular/core';
import { MatSnackBar } from '@angular/material/snack-bar';
import { Router } from '@angular/router';
import { Problem } from '../problem';
import { ProblemService } from '../problem-service/problem.service';

@Component({
  selector: 'app-student-solution-component',
  templateUrl: './student-solution-component.component.html',
  styleUrls: ['./student-solution-component.component.css']
})
export class StudentSolutionComponentComponent implements OnInit{
  constructor(private problemService: ProblemService, private router: Router, private _snackBar: MatSnackBar) {}
  problem: Problem;

  ngOnInit(): void {
      this.problem = this.problemService.getSelectedProblem();
      if(!this.problem){
        this._snackBar.open('No problem selected','X', {duration: 2000});
        this.router.navigate(['problem-selection']);
      }
  }

  
}
